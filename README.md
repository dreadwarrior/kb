## rbenv usage instructions

    # init rbenv in current shell session
    eval $(rbenv init - bash)
    # check ruby version (match .ruby-version)
    ruby -v
    # export gem path
    export PATH="$HOME/.local/share/gem/ruby/3.0.0/bin:$PATH"
